qb <- qbrun |> filter(RushYds > 500 & RushTDs > 5 & RushAtt > 75)

Player <- qb |> filter (Player == "Tim Tebow" | Player =="Cam Newton")
  
ggplot() + 
  geom_point(
    data=qb, 
    aes(x=RushYds, y=RushAtt, size=RushTDs),
color="grey",
alpha=.5) +
    geom_point(
      data=Player,
      aes(RushYds, y=RushAtt, size=RushTDs), color="blue"
    ) +
  labs(
    title = "Cam Newton & Tim Tebow Get In The Endzone",
    x = "Rush Yards",
    y = "Rush Attempts",
    size = "Touchdowns"
  ) +
  geom_vline(xintercept = 5.863151) + 
  geom_hline(yintercept = 5.448228) +
geom_text_repel(
    data=Player, 
    aes(x=RushYds, y=RushAtt, label=Player)
  ) +
